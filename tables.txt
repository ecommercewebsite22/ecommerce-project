CREATE TABLE user_info(
  user_id SERIAL PRIMARY KEY AUTO,
  username varchar(50),
  password varchar(50),
  first_name varchar(20),
  last_name varchar(20),
  email varchar(50),
  user_type varchar(10)
);

CREATE TABLE products(
  product_sku SERIAL PRIMARY KEY AUTO,
  product_name varchar(30),
  product_image varchar(255),
  product_quantity INT,
  product_price NUMERIC
);

CREATE TABLE product_details(
  details_description varchar(255),
  details_images varchar(255),
);

CREATE TABLE history(
  cart_no SERIAL PRIMARY KEY AUTO,
  user_id INT FOREIGN KEY REFERENCES user_info(user_id)
  date TIMESTAMP
);

CREATE TABLE cart(
  cart_subID SERIAL PRIMARY KEY AUTO REFERENCES cart_history(cart_no),
  product_sku INT REFERENCES products(product_sku),
  cart_quantity INT,
  product_price NUMERIC REFERENCES products(product_price)
);
